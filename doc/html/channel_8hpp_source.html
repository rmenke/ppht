154static inline std::unique_ptr<scanner> make_scanner(segment_t &segment) {
155 auto &a = std::get<0>(segment);
156 auto &b = std::get<1>(segment);
157
158 auto &x0 = std::get<0>(a);
159 auto &y0 = std::get<1>(a);
160 auto &x1 = std::get<0>(b);
161 auto &y1 = std::get<1>(b);
162
163 const auto dx = std::abs(x0 - x1);
164 const auto dy = std::abs(y0 - y1);
165
166 if (dx >= dy) {
167 if (x0 > x1) {
168 std::swap(a, b);
169 }
170
171 if (y0 < y1) {
172 return std::unique_ptr<scanner>(
173 new bresenham_scanner<0, +1>{x0, y0, x1, y1});
174 }
175 else if (y0 == y1) {
176 return std::unique_ptr<scanner>(new axis_scanner<0>{});
177 }
178 else { // y0 > y1
179 return std::unique_ptr<scanner>(
180 new bresenham_scanner<0, -1>{x0, y0, x1, y1});
181 }
182 }
183 else {
184 if (y0 > y1) {
185 std::swap(a, b);
186 }
187
188 if (x0 < x1) {
189 return std::unique_ptr<scanner>(
190 new bresenham_scanner<1, +1>{x0, y0, x1, y1});
191 }
192 else if (x0 == x1) {
193 return std::unique_ptr<scanner>(new axis_scanner<1>{});
194 }
