203static inline std::set<point_t>
204find_offsets(segment_t const &segment, unsigned radius) {
205 // The vector [xn, yn] is normal to the segment.
206
207 double xn = std::get<1>(segment.second);
208 xn -= std::get<1>(segment.first);
209 double yn = std::get<0>(segment.first);
210 yn -= std::get<0>(segment.second);
211
212 // Make [xn, yn] the unit normal vector.
213
214 double len = std::hypot(xn, yn);
215 xn /= len, yn /= len;
216
217 std::set<point_t> result{{0, 0}};
218
219 for (auto r = 1U; r <= radius; ++r) {
220 auto dx = std::lround(xn * r);
221 auto dy = std::lround(yn * r);
222
223 result.emplace(dx, dy);
224 result.emplace(-dx, -dy);
225 }
226
227 return result;
228}

257template <template <class> class Raster>
258point_set scan(state<Raster> &s, segment_t const &segment, unsigned radius,
259 unsigned max_gap) {
260 auto const offsets = find_offsets(segment, radius);
261
262 // The initial gap is technically infinite, but anything
263 // larger than max_gap will do.
264 auto gap = max_gap + 1;
265
266 std::vector<point_set> point_sets;
267
268 for (auto const &point : channel(segment)) {
269 std::set<point_t> points;
270
271 for (auto const &offset : offsets) {
272 auto p = point + offset;
273
274 if (p[0] < 0 || p[0] >= static_cast<long>(s.cols())) continue;
275 if (p[1] < 0 || p[1] >= static_cast<long>(s.rows())) continue;
276
277 auto status = s.status(p);
278
279 if (status == status_t::pending || status == status_t::voted) {
280 points.emplace(p);
281 }
282 }
283
284 if (points.empty()) { // no hits
285 ++gap;
286 }
287 else {
288 // If the gap is too large to ignore, start a new point set.
289 if (gap > max_gap) point_sets.emplace_back();
290 point_sets.back().add_point(point, points);
291
292 gap = 0;
293 }
294 }
295
296 if (point_sets.empty()) {
297 throw std::logic_error{"channel contained no viable segments"};
298 }
299
300 auto longest = std::max_element(point_sets.begin(), point_sets.end());
301
302 return *longest;
303}
