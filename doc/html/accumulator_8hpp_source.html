196 static std::pair<std::size_t, int>
197 rho_info(std::size_t rows, std::size_t cols,
198 std::size_t max_theta) noexcept {
199 double diag = std::ceil(std::hypot(rows - 1, cols - 1));
200 int rho_exp = std::ilogb(max_theta / (diag * 2.0 + 1.0));
201
202 // lo is 2 * diag * (1 << rho_exp).
203
204 std::size_t lo = std::ceil(std::scalbn(diag, rho_exp + 1)) + 1;
205
206 // hi is 2 * diag * (2 << rho_exp). This is the smallest
207 // rho_exp such that 2 * diag * (1 << rho_exp) >= max_theta.
208
209 std::size_t hi = std::ceil(std::scalbn(diag, rho_exp + 2)) + 1;
210
211 assert(lo <= max_theta && max_theta <= hi);
212
213 // Select the rho_exp that makes the counter matrix the most
214 // square.
215
216 if ((max_theta - lo) <= (hi - max_theta)) {
217 return std::make_pair(lo, rho_exp);
218 }
219 else {
220 return std::make_pair(hi, rho_exp + 1);
221 }
222 }

257 segment_t find_segment(std::size_t theta, double rho) {
258 // There are a few degenerate cases where multiple matches for
259 // the same endpoint can be found, e.g., a line through the
260 // origin. Using a set eliminates most of these cases. See
261 // the comment at the end for what happens to those that slip
262 // through.
263
264 std::set<point_t> endpoints;
265
266 auto const &cossin = _trig[theta];
267 auto const &sin_theta = std::get<1>(cossin);
268 auto const &cos_theta = std::get<0>(cossin);
269
270 auto get_x = [&](double y) -> long {
271 double x = std::rint((rho - sin_theta * y) / cos_theta);
272 return clamp<long>(x);
273 };
274
275 auto get_y = [&](double x) -> long {
276 double y = std::rint((rho - cos_theta * x) / sin_theta);
277 return clamp<long>(y);
278 };
279
280 auto x_min = get_x(0);
281 auto y_min = get_y(0);
282 auto x_max = get_x(_rows - 1);
283 auto y_max = get_y(_cols - 1);
284
285 if (0 <= y_min && y_min < static_cast<int>(_rows)) {
286 endpoints.emplace(0, y_min);
287 }
288 if (0 <= x_min && x_min < static_cast<int>(_cols)) {
289 endpoints.emplace(x_min, 0);
290 }
291 if (0 <= y_max && y_max < static_cast<int>(_rows)) {
292 endpoints.emplace(_cols - 1, y_max);
293 }
294 if (0 <= x_max && x_max < static_cast<int>(_cols)) {
295 endpoints.emplace(x_max, _rows - 1);
296 }
297
298 if (endpoints.empty()) {
299 throw std::logic_error{"line (" + std::to_string(theta) + ", " +
300 std::to_string(rho) +
301 ") does not intersect bitmap"};
302 }
303
304 // If endpoints.size() > 2, ignore the points in the middle.
305 // If endpoints.size() == 1, create single-pixel segment.
306
307 return segment_t{*endpoints.begin(), *endpoints.rbegin()};
308 }

327 bool vote(point_t const &p, segment_t &segment) {
328 auto const max_rho = _counters.rows();
329 auto const max_theta = _counters.cols();
330
331 Count n = _min_trigger_points;
332
333 using line_t = std::pair<std::size_t, double>;
334
335 std::vector<line_t> found;
336
337 std::size_t theta;
338 double rho;
339
340 // Increment the cells in the register, keeping track of the
341 // current maxima.
342
343 for (theta = 0; theta < max_theta; ++theta) {
344 rho = scale_rho(p.dot(_trig[theta]));
345 if (rho < 0 || rho >= max_rho) continue;
346
347 auto &counter = _counters[rho][theta];
348
349 ++counter;
350
351 // If we have found a new maxima, update n and discard the
352 // old candidates.
353
354 if (n < counter) {
355 n = counter;
356 found.clear();
357 }
358
359 if (n == counter) {
360 found.emplace_back(theta, unscale_rho(rho));
361 }
362 }
363
364 ++_votes;
365
366 // If we do not have a candidate line, stop.
367
368 if (found.empty()) return false;
369
370 // There are max_theta * max_rho cells in the register.
371 // Each vote will increment max_theta of these cells, one
372 // per column.
373 //
374 // Assuming the null hypothesis (the image is random noise),
375 // E[n] = votes/max_rho for all cells in the register.
376
377 double const lambda = static_cast<double>(_votes) / max_rho;
378
379 // For the null hypothesis, the cells are filled (roughly)
380 // according to a Poisson model:
381 //
382 // p(n) = λⁿ/n!·exp(-λ)
383 // = λⁿ/Γ(n+1)·exp(-λ)
384 // ln p(n) = n·ln(λ) - lnΓ(n+1) - λ
385
386 double const lnp =
387 n * std::log(lambda) - std::lgamma(n + 1) - lambda;
388
389 // lnp is the log probability that a bin filled randomly would
390 // contain a count of n. If the probability is above the
391 // significance threshold, we assume that the bin was filled
392 // by noise and tell the caller we did not find a segment.
393
394 if (lnp >= _log_threshold) return false;
395
396 // Reject the null hypothesis.
397
398 if (found.size() == 1) {
399 std::tie(theta, rho) = found.at(0);
400 }
401 else {
402 auto begin = found.begin();
403 auto end = found.end();
404
405 if (begin->first == 0 && (end-1)->first == max_theta - 1) {
406 // If we have values at both ends of the vector there
407 // is a chance that we have a "wraparound" cluster.
408 // Since every line (θ, ρ) has a counterpart of (θ +
409 // π/2, -ρ), replace the points at the beginning of
410 // the vector with their counterparts, effectively
411 // transplanting the points with angles in quadrant I
412 // into points with angles in quadrant III.
413
414 for (auto iter = begin; iter != end; ++iter) {
415 if (iter->first < max_theta / 2) {
416 iter->first += max_theta;
417 iter->second = -(iter->second);
418 }
419 else {
420 break;
421 }
422 }
423
424 std::sort(begin, end);
425 }
426
427 // Recalculate the maximum distance between the first and
428 // last in the cluster.
429
430 auto const diff = (end-1)->first - begin->first + 1;
431
432 // If the difference is the same as the number of points
433 // in the cluster, then the range is continuous and it is
434 // safe to take the median value. Otherwise, return.
435
436 if (diff != found.size()) return false;
437
438 auto median = begin + diff / 2;
439 std::tie(theta, rho) = *median;
440
441 // If the median is in quadrant III, move it back to
442 // quadrant I.
443
444 if (theta >= max_theta) {
445 theta -= max_theta, rho = -rho;
446 }
447 }
448
449 segment = find_segment(theta, rho);
450 return true;
451 }

458 void unvote(point_t const &p) {
459 auto const max_rho = _counters.rows();
460 auto const max_theta = _counters.cols();
461
462 for (std::size_t theta = 0; theta < max_theta; ++theta) {
463 double const rho = scale_rho(p.dot(_trig[theta]));
464 if (rho < 0 || rho >= max_rho) continue;
465
466 auto &counter = _counters[rho][theta];
467
468 if (counter == 0) {
469 throw std::logic_error{__func__};
470 }
471
472 --counter;
473 }
474
475 --_votes;
476 }
