197  rho_info(std::size_t rows, std::size_t cols,
198  std::size_t max_theta) noexcept {
199  double diag = std::ceil(std::hypot(rows - 1, cols - 1));
200  int rho_exp = std::ilogb(max_theta / (diag * 2.0 + 1.0));
201 
202  // lo is 2 * diag * (1 << rho_exp).
203 
204  std::size_t lo = std::ceil(std::scalbn(diag, rho_exp + 1)) + 1;
205 
206  // hi is 2 * diag * (2 << rho_exp). This is the smallest
207  // rho_exp such that 2 * diag * (1 << rho_exp) >= max_theta.
208 
209  std::size_t hi = std::ceil(std::scalbn(diag, rho_exp + 2)) + 1;
210 
211  assert(lo <= max_theta && max_theta <= hi);
212 
213  // Select the rho_exp that makes the counter matrix the most
214  // square.
215 
216  if ((max_theta - lo) <= (hi - max_theta)) {
217  return std::make_pair(lo, rho_exp);
218  }
219  else {
220  return std::make_pair(hi, rho_exp + 1);
221  }
222  }

257  segment_t find_segment(std::size_t theta, double rho) {
258  // There are a few degenerate cases where multiple matches for
259  // the same endpoint can be found, e.g., a line through the
260  // origin. Using a set eliminates most of these cases. See
261  // the comment at the end for what happens to those that slip
262  // through.
263 
264  std::set<point_t> endpoints;
265 
266  auto const &cossin = _trig[theta];
267  double const &sin_theta = cossin[1];
268  double const &cos_theta = cossin[0];
269 
270  auto get_x = [&](double y) -> long {
271  double x = std::rint((rho - sin_theta * y) / cos_theta);
272  return clamp<long>(x);
273  };
274 
275  auto get_y = [&](double x) -> long {
276  double y = std::rint((rho - cos_theta * x) / sin_theta);
277  return clamp<long>(y);
278  };
279 
280  auto x_min = get_x(0);
281  auto y_min = get_y(0);
282  auto x_max = get_x(_rows - 1);
283  auto y_max = get_y(_cols - 1);
284 
285  if (0 <= y_min && y_min < static_cast<int>(_rows)) {
286  endpoints.emplace(0, y_min);
287  }
288  if (0 <= x_min && x_min < static_cast<int>(_cols)) {
289  endpoints.emplace(x_min, 0);
290  }
291  if (0 <= y_max && y_max < static_cast<int>(_rows)) {
292  endpoints.emplace(_cols - 1, y_max);
293  }
294  if (0 <= x_max && x_max < static_cast<int>(_cols)) {
295  endpoints.emplace(x_max, _rows - 1);
296  }
297 
298  if (endpoints.empty()) {
299  throw std::logic_error{"line (" + std::to_string(theta) + ", " +
300  std::to_string(rho) +
301  ") does not intersect bitmap"};
302  }
303 
304  // If endpoints.size() > 2, ignore the points in the middle.
305  // If endpoints.size() == 1, create single-pixel segment.
306 
307  return segment_t{*endpoints.begin(), *endpoints.rbegin()};
308  }

327  bool vote(point_t const &p, segment_t &segment) {
328  auto const max_rho = _counters.rows();
329  auto const max_theta = _counters.cols();
330 
331  Count n = _min_trigger_points;
332 
333  using line_t = std::pair<std::size_t, double>;
334 
335  std::vector<line_t> found;
336 
337  std::size_t theta;
338  double rho;
339 
340  for (theta = 0; theta < max_theta; ++theta) {
341  rho = scale_rho(p.dot(_trig[theta]));
342  if (rho < 0 || rho >= max_rho) continue;
343 
344  auto &counter = _counters[rho][theta];
345 
346  ++counter;
347 
348  if (n < counter) {
349  n = counter;
350  found.clear();
351  }
352  if (n == counter) {
353  found.emplace_back(theta, unscale_rho(rho));
354  }
355  }
356 
357  ++_votes;
358 
359  // If we do not have a candidate line, stop.
360 
361  if (found.empty()) return false;
362 
363  // There are max_theta * max_rho cells in the register.
364  // Each vote will increment max_theta of these cells, one
365  // per column.
366  //
367  // Assuming the null hypothesis (the image is random noise),
368  // E[n] = votes/max_rho for all cells in the register.
369 
370  double const lambda = static_cast<double>(_votes) / max_rho;
371 
372  // For the null hypothesis, the cells are filled (roughly)
373  // according to a Poisson model:
374  //
375  // p(n) = λⁿ/n!·exp(-λ)
376  // = λⁿ/Γ(n+1)·exp(-λ)
377  // ln p(n) = n·ln(λ) - lnΓ(n+1) - λ
378 
379  double const lnp =
380  n * std::log(lambda) - std::lgamma(n + 1) - lambda;
381 
382  // lnp is the log probability that a bin filled randomly would
383  // contain a count of n. If the probability is above the
384  // significance threshold, we assume that the bin was filled
385  // by noise and tell the caller we did not find a segment.
386 
387  if (lnp >= _log_threshold) return false;
388 
389  // Reject the null hypothesis.
390 
391  if (found.size() == 1) {
392  std::tie(theta, rho) = found.at(0);
393  }
394  else {
395  auto begin = found.begin();
396  auto end = found.end();
397 
398  if (begin->first == 0 && (end-1)->first == max_theta - 1) {
399  // If we have values at both ends of the vector there
400  // is a chance that we have a "wraparound" cluster.
401  // Since every line (θ, ρ) has a counterpart of (θ +
402  // π/2, -ρ), replace the points at the beginning of
403  // the vector with their counterparts, effectively
404  // transplanting the points with angles in quadrant I
405  // into points with angles in quadrant III.
406 
407  for (auto iter = begin; iter != end; ++iter) {
408  if (iter->first < max_theta / 2) {
409  iter->first += max_theta;
410  iter->second = -(iter->second);
411  }
412  else {
413  break;
414  }
415  }
416 
417  std::sort(begin, end);
418  }
419 
420  // Recalculate the maximum distance between the first and
421  // last in the cluster.
422 
423  auto const diff = (end-1)->first - begin->first + 1;
424 
425  // If the difference is the same as the number of points
426  // in the cluster, then the range is continuous and it is
427  // safe to take the median value. Otherwise, return.
428 
429  if (diff != found.size()) return false;
430 
431  auto median = begin + diff / 2;
432  std::tie(theta, rho) = *median;
433 
434  // If the median is in quadrant III, move it back to
435  // quadrant I.
436 
437  if (theta >= max_theta) {
438  theta -= max_theta, rho = -rho;
439  }
440  }
441 
442  segment = find_segment(theta, rho);
443  return true;
444  }
